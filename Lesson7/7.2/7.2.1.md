#### ~ tutorialhint

```javascript
let pointsEarned =  0

let totalScore =  0

blocks.onBlockPlaced(GRASS,  function  ()  {
	pointsEarned +=  1
})

blocks.onBlockBroken(GRASS,  function  ()  {
	pointsLost +=  1
})
  
function checkScore ()  {
	totalScore = pointsEarned - pointsLost
	return totalScore
}

player.onChat("score",  function  ()  {
	checkScore()
	player.say(totalScore)
})
```
